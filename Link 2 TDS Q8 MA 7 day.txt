Q7.  Write an SQL query to compute moving average of how 
much customer paid in a 7 days 
window (current day + 6 days before)



select visited_on, SUM(amount) OVER( order by visited_on ROWS BETWEEN 6 PRECEEDING AND CURRENT ROW), ROUND(AVG(amount) OVER(order by visited_on ROWS BETWEEN 6 PRECEEDING AND CURRENT ROW),2) as average_amount from
(
select visited_on, sum(amount) as amount from customer
				group by visited_on
)T1

order by visited_on OFFSET 6
