Cancellation rates
From the following table of user IDs, actions, and dates, write a query to return the publication and cancellation rate for each user.

1.
select user_id, publish/starts as publish_rate, cancels/starts as cancel rate from
(select user_id,
	SUM(case when action='start' then 1 else 0 END ) as starts
	SUM(case when action='publish' then 1 else 0 END ) as publish
	SUM(case when action='cancel' then 1 else 0 END ) as cancels
	group by user_id) T1

2. 
select user_id, sum(publish)/sum(starts) as publish_rate, sum(cancel)/sum(starts) as cancel_rate

from
(
select user_id, (case when action=publish then 1 else 0 end) as publish,
		(case when action=cancel then 1 else 0 end) as cancel ,
		(case when action=publish then 1 else 0 end) as start from users
)T1
group by user_id

3. (easier)
Select user_id, sum(case when action=‘publish’ then 1 else 0 end) / sum(case when action=‘start’ then 1 else 0 end) as publish_rate,
sum(case when action=‘cancel’ then 1 else 0 end) / sum(case when action=‘start’ then 1 else 0 end) as cancel_rate
Group by 1

Python

df_starts=df.loc[df['action']=='start',:].groupby(['user_id'], as_index=False).agg(start_count=('action', 'count'))  --- This returns a df with two columns user_id and start_count

df_publish=df.loc[df['action']=='publish',:].groupby(['user_id'], as_index=False).agg(publish_count=('action', 'count'))
df_cancel=df.loc[df['action']=='cancel',:].groupby(['user_id'], as_index=False).agg(cancel_count=('action', 'count')) 

df_join=df_starts.merge(df_publish, how='inner', on='user_id')
df_join_final=df_join.merge(df_cancel, how='inner', on='user_id')
df_join_final['publish_rate']=df_join_final['publish_count]/df_join_final['start_count]    --- This is accurate
df_join_final['cancel_rate']=df_join_final['cancel_count']/df_join_final['start_count]

df_join_final[:,['publish_rate','cancel_rate']
