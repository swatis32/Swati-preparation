7. Mobile and web visitors
With the following two tables, return the fraction of users who only visited mobile, only visited web, and visited both.

select sum(mob)/count(*), sum(web)/count(*), sum(both)/cout(*) from
(
select user_id, coalesce(mob,0) as 0, coalesce(web,0) as web, case when mob=1 and web=1 then 1 
									else 0 end
							       as both
from

(select user_id, count(Distinct user_id) as mob from mobile
group by user_id) T1

full outer join
(
select user_id, count(Distinct user_id) as wb from web
group by user_id
)T2
)T3
